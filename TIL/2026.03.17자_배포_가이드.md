### 2-2. QRTZ_CRON_TRIGGERS 스케줄 변경

```sql
-- ============================================================
-- 배치 스케줄 Cron 표현식 변경 (총 10개)
-- ============================================================

-- [partnerGroup]
-- partnerDeliveryFeeTrigger: 10시,13시 → 9시,10시
UPDATE QRTZ_CRON_TRIGGERS
SET CRON_EXPRESSION = '0 0 9,10 * * ?'
WHERE TRIGGER_NAME = 'partnerDeliveryFeeTrigger'
  AND TRIGGER_GROUP = 'partnerGroup';

-- partnerDetailTrigger: 30분마다 → 매일 14시
UPDATE QRTZ_CRON_TRIGGERS
SET CRON_EXPRESSION = '0 0 14 * * ?'
WHERE TRIGGER_NAME = 'partnerDetailTrigger'
  AND TRIGGER_GROUP = 'partnerGroup';

-- [riderGroup]
-- riderInfoTrigger: 30분마다 → 매시 정각
UPDATE QRTZ_CRON_TRIGGERS
SET CRON_EXPRESSION = '0 0 * * * ?'
WHERE TRIGGER_NAME = 'riderInfoTrigger'
  AND TRIGGER_GROUP = 'riderGroup';

-- [deliveryGroup]
-- deliveryDailyHistoriesTrigger: 30분마다 → 9시,10시
UPDATE QRTZ_CRON_TRIGGERS
SET CRON_EXPRESSION = '0 0 9,10 * * ?'
WHERE TRIGGER_NAME = 'deliveryDailyHistoriesTrigger'
  AND TRIGGER_GROUP = 'deliveryGroup';

-- riderDeliveryHistoriesTrigger: 30분마다 → 9시,10시
UPDATE QRTZ_CRON_TRIGGERS
SET CRON_EXPRESSION = '0 0 9,10 * * ?'
WHERE TRIGGER_NAME = 'riderDeliveryHistoriesTrigger'
  AND TRIGGER_GROUP = 'deliveryGroup';

-- [migrationGroup]
-- riderInfoIfTrigger: 10분마다 → 매시 5분
UPDATE QRTZ_CRON_TRIGGERS
SET CRON_EXPRESSION = '0 5 * * * ?'
WHERE TRIGGER_NAME = 'riderInfoIfTrigger'
  AND TRIGGER_GROUP = 'migrationGroup';

-- partnerDetailsIfTrigger: 10분마다 → 매시 5분
UPDATE QRTZ_CRON_TRIGGERS
SET CRON_EXPRESSION = '0 5 * * * ?'
WHERE TRIGGER_NAME = 'partnerDetailsIfTrigger'
  AND TRIGGER_GROUP = 'migrationGroup';

-- partnerDeliveryFeesIfTrigger: 10분마다 → 매시 5분
UPDATE QRTZ_CRON_TRIGGERS
SET CRON_EXPRESSION = '0 5 * * * ?'
WHERE TRIGGER_NAME = 'partnerDeliveryFeesIfTrigger'
  AND TRIGGER_GROUP = 'migrationGroup';

-- riderDeliveryHistoriesIfTrigger: 10분마다 → 매시 5분
UPDATE QRTZ_CRON_TRIGGERS
SET CRON_EXPRESSION = '0 5 * * * ?'
WHERE TRIGGER_NAME = 'riderDeliveryHistoriesIfTrigger'
  AND TRIGGER_GROUP = 'migrationGroup';

-- [batchGroup]
-- batchWithdrawalTrigger: 매시 정각 → 매일 00:05
UPDATE QRTZ_CRON_TRIGGERS
SET CRON_EXPRESSION = '0 5 0 * * ?'
WHERE TRIGGER_NAME = 'batchWithdrawalTrigger'
  AND TRIGGER_GROUP = 'batchGroup';
```

**확인 쿼리:**
```sql
SELECT TRIGGER_NAME, TRIGGER_GROUP, CRON_EXPRESSION
FROM QRTZ_CRON_TRIGGERS
ORDER BY TRIGGER_GROUP, TRIGGER_NAME;
```

## 참고 사항

- **SCHED_NAME**: 운영 DB에서는 `schedulerFactoryBean` 사용 (application.yml과 다를 수 있음)
- **NEXT_FIRE_TIME**: 수동 INSERT 시 NULL로 설정해도 Quartz 재시작 시 자동 계산
- **외래키 순서**:
  - INSERT: `QRTZ_JOB_DETAILS` → `QRTZ_TRIGGERS` → `QRTZ_CRON_TRIGGERS`
  - DELETE: `QRTZ_CRON_TRIGGERS` → `QRTZ_TRIGGERS` → `QRTZ_JOB_DETAILS`
